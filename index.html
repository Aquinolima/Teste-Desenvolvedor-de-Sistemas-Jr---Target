Desafio 2: Verificar se número pertence à sequência de Fibonacci

function verificaFibonacci(num) {
    let a = 0, b = 1;
    while (a < num) {
        let temp = a;
        a = b;
        b = temp + b;
    }
    return a === num;
}

let numero = 8;
console.log(`O número ${numero} pertence à sequência de Fibonacci.`);
Resultado: O número 8 pertence à sequência de Fibonacci.

Desafio 5: Inversão de string

function inverterString(s) {
    let sInvertida = "";
    for (let i = s.length - 1; i >= 0; i--) {
        sInvertida += s[i];
    }
    return sInvertida;
}

let entrada = "Thiago";
console.log("String invertida: " + inverterString(entrada));
Resultado: ogaihT
